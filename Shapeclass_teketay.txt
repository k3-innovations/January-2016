package shape;

public class Shape {

	private int numberOfSide;
	private String color;

// draw method which assign the no. sides of the polygon
	public void draw(int side) {
		numberOfSide = side;

	}
// filling method which assign color string
	public void fillColor(String col)

	{
		color = col;
	}

	// toString is overrided
	public String toString() {

 return "The number of sides of the polygon is" + " " + 
 numberOfSide + " " + "and filled with" + " " + color+ " " + "Color";
	}
}
// square class 
public class Square extends Shape {


}

public class Hexagon extends Shape {
	

}
public class DrawShapes {

public static void main(String[] args) {
	Hexagon hex = new Hexagon();
	Square sqr =new Square();
	hex.draw(6);
	hex.fillColor("blue");
	sqr.draw(4);
	sqr.fillColor("red");
	System.out.println(sqr);
	System.out.println(hex);
}
}
